```python
def execute_parallel(self, plan, context):
    """执行 plan,无依赖步骤并行,有依赖步骤串行。

    算法:
    1. 解析每步的依赖(扫描 params 中的 {{stepN.output...}} 引用)
    2. 用拓扑排序分组:无依赖的可并行,有依赖的串行
    3. 每组用 asyncio.gather 并行执行
    """
    import asyncio
    from .plan_serializer import Plan, PlanStep

    # 1. 解析依赖
    steps = plan.steps
    dep_graph = self._build_dependency_graph(steps)

    # 2. 拓扑分组
    groups = self._topological_groups(steps, dep_graph)

    # 3. 按组执行
    step_results = {}
    all_output = []
    for group in groups:
        if len(group) == 1:
            # 单步:串行
            step = group[0]
            idx = steps.index(step)
            result = self._execute_step_with_strategy(
                step, f"step{idx + 1}", step_results, context, idx
            )
            result["latency_ms"] = int(time.time() * 1000) % 100  # 简化
            step_results[f"step{idx + 1}"] = result
            all_output.append(result)
        else:
            # 多步:并行
            async def run_one(step, idx):
                return self._execute_step_with_strategy(
                    step, f"step{idx + 1}", step_results, context, idx
                )

            async def run_group():
                tasks = [run_one(step, steps.index(step)) for step in group]
                return await asyncio.gather(*tasks)

            group_results = asyncio.run(run_group())
            for i, step in enumerate(group):
                idx = steps.index(step)
                result = group_results[i]
                result["latency_ms"] = int(time.time() * 1000) % 100
                step_results[f"step{idx + 1}"] = result
                all_output.append(result)

    return all_output


def _build_dependency_graph(self, steps):
    """扫描每个 step 的 params,提取 {{stepN.output...}} 引用作为依赖。"""
    import re
    dep_pattern = re.compile(r"\{\{step(\d+)\.")
    graph = {i: set() for i in range(len(steps))}
    for i, step in enumerate(steps):
        params_str = str(step.params)
        for m in dep_pattern.finditer(params_str):
            dep_step = int(m.group(1)) - 1
            if 0 <= dep_step < i:  # 只能依赖前序步骤
                graph[i].add(dep_step)
    return graph


def _topological_groups(self, steps, dep_graph):
    """拓扑排序分组:同一层的可并行。"""
    in_degree = {i: len(dep_graph[i]) for i in range(len(steps))}
    groups = []
    remaining = set(range(len(steps)))

    while remaining:
        # 找当前无依赖的步骤
        current = sorted([i for i in remaining if in_degree[i] == 0])
        if not current:
            # 有环(不应该),全打包
            current = sorted(remaining)
        groups.append([steps[i] for i in current])

        # 更新 in_degree
        for i in current:
            remaining.discard(i)
            for j in remaining:
                if i in dep_graph[j]:
                    in_degree[j] -= 1
    return groups
```

```python
"""缓存击穿防护测试 — singleflight 单飞模式。

Task 3 of feat/sa-perf-ux: 同一时刻同 key 只查一次 DB/LLM。
"""
import threading
import time
from unittest.mock import patch

import pytest

from omni_desk_backend.smart_assistant import cache as cache_module
from omni_desk_backend.smart_assistant.cache import singleflight_get_or_set


class TestSingleflight:
    def test_concurrent_calls_only_execute_loader_once(self):
        """50 个并发同 key 调用,loader 只执行 1 次。"""
        call_count = {"n": 0}
        call_lock = threading.Lock()

        def slow_loader():
            with call_lock:
                call_count["n"] += 1
            time.sleep(0.1)
            return "loaded_value"

        def worker():
            return singleflight_get_or_set("test_key", slow_loader, ttl=60)

        threads = [threading.Thread(target=worker) for _ in range(50)]
        for t in threads:
            t.start()
        for t in threads:
            t.join()
        # 直接读 cache
        assert call_count["n"] == 1, f"loader 期望调用 1 次,实际 {call_count['n']} 次"

    def test_sequential_calls_each_invoke_loader(self):
        """顺序调用,cache 过期后 loader 重新执行。"""
        call_count = {"n": 0}

        def loader():
            call_count["n"] += 1
            return f"v{call_count['n']}"

        v1 = singleflight_get_or_set("seq_key", loader, ttl=60)
        assert v1 == "v1"
        # cache hit,不调 loader
        v2 = singleflight_get_or_set("seq_key", lambda: "v_new", ttl=60)
        assert v2 == "v1"
        assert call_count["n"] == 1
```

```python
import threading

# singleflight: 防止缓存击穿(同 key 并发时只调一次 loader)
_inflight_locks: dict[str, threading.Lock] = {}
_inflight_values: dict[str, object] = {}
_inflight_flags: dict[str, threading.Event] = {}
_inflight_global = threading.Lock()


def singleflight_get_or_set(key: str, loader, ttl: int = 7200):
    """缓存击穿防护:同 key 并发时只调一次 loader。

    Args:
        key: 缓存 key(应已含 cache_version + context_sig)
        loader: 缓存未命中时调用的函数
        ttl: 缓存 TTL(秒)

    Returns:
        缓存值
    """
    # 1. 先查 cache
    cached = cache.get(_key("sf", key))
    if cached is not None:
        return cached

    # 2. 缓存未命中,看是否有人正在加载
    with _inflight_global:
        if key in _inflight_flags:
            # 已有线程在加载,等待
            event = _inflight_flags[key]
            is_leader = False
        else:
            # 当前线程负责加载
            event = threading.Event()
            _inflight_flags[key] = event
            is_leader = True

    if not is_leader:
        # 其他线程在加载,等待
        event.wait(timeout=30)
        return cache.get(_key("sf", key))

    # 当前线程加载
    try:
        value = loader()
        cache.set(_key("sf", key), value, ttl)
        _inflight_values[key] = value
        return value
    finally:
        # 唤醒所有等待线程
        event.set()
        with _inflight_global:
            _inflight_flags.pop(key, None)
            _inflight_values.pop(key, None)
```

```python
from django.test import override_settings

from omni_desk_backend.smart_assistant.cache import (

class TestCacheVersion:
    def test_cache_key_includes_version(self):
```

```python
from django.conf import settings


def _build_cache_key(query: str, user_id: int, intent: str) -> str:
    """构建缓存键,包含 cache_version。"""
    import hashlib
    cache_version = getattr(settings, "SMART_ASSISTANT_CACHE_VERSION", "1.0")
    raw_key = f"{query}|{user_id}|{cache_version}|{intent}"
    return hashlib.sha256(raw_key.encode()).hexdigest()


def cache_answer(query: str, intent: str, answer: str, context_sig: str, ttl: int = 7200):
    """缓存回答,键包含 cache_version。"""
    # 从 context_sig 提取 user_id(简化)
    user_id = int(context_sig.split("_")[0].replace("u", "")) if "u" in context_sig else 0
    key = _build_cache_key(query=query, user_id=user_id, intent=intent)
    cache.set(key, answer, ttl)


def get_cached_answer(query: str, intent: str, context_sig: str):
    """获取缓存的回答,键包含 cache_version。"""
    user_id = int(context_sig.split("_")[0].replace("u", "")) if "u" in context_sig else 0
    key = _build_cache_key(query=query, user_id=user_id, intent=intent)
    return cache.get(key)
```

```javascript
import { useState, useEffect, useRef } from 'react';

function SmartChatPage() {
  const [chunks, setChunks] = useState([]);
  const [displayedChunks, setDisplayedChunks] = useState([]);
  const [isStreaming, setIsStreaming] = useState(false);
  const animationFrameRef = useRef(null);
  const lastUpdateRef = useRef(0);

  // 打字机效果:每 50ms 显示一个 chunk
  useEffect(() => {
    if (!isStreaming || chunks.length === 0) return;

    const animate = (timestamp) => {
      if (timestamp - lastUpdateRef.current >= 50) {
        setDisplayedChunks((prev) => {
          const nextIndex = prev.length;
          if (nextIndex < chunks.length) {
            return [...prev, chunks[nextIndex]];
          }
          return prev;
        });
        lastUpdateRef.current = timestamp;
      }
      animationFrameRef.current = requestAnimationFrame(animate);
    };

    animationFrameRef.current = requestAnimationFrame(animate);

    return () => {
      if (animationFrameRef.current) {
        cancelAnimationFrame(animationFrameRef.current);
      }
    };
  }, [isStreaming, chunks]);

  // SSE 流式处理
  const handleStream = async (query) => {
    setIsStreaming(true);
    setChunks([]);
    setDisplayedChunks([]);

    const response = await fetch('/api/smart-assistant/chat/stream/', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ query }),
    });

    const reader = response.body.getReader();
    const decoder = new TextDecoder();

    while (true) {
      const { done, value } = await reader.read();
      if (done) break;

      const text = decoder.decode(value);
      const lines = text.split('\n').filter((line) => line.startsWith('data: '));

      for (const line of lines) {
        const data = JSON.parse(line.replace('data: ', ''));
        if (data.type === 'chunk') {
          setChunks((prev) => [...prev, data.content]);
        }
      }
    }

    setIsStreaming(false);
  };

  // 渲染:使用 displayedChunks 而非 chunks
  const displayedText = displayedChunks.join('');

  return (
    <div>
      {/* 聊天内容 */}
      <div>{displayedText}</div>
      {/* 输入框 + 发送按钮 */}
    </div>
  );
}
```
